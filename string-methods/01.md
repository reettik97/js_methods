Example:
```js
  // your code goes here

var stringParameterize = (str) =>{
  var stringPara = '';
  str = str.trim();
  var separateTheWords = str.split(' ');
  stringPara = separateTheWords[0].toLowerCase();
  for(let i=1;i<separateTheWords.length;i++){
    if(separateTheWords[i]){
      stringPara += '-' + separateTheWords[i].toLowerCase();
```
